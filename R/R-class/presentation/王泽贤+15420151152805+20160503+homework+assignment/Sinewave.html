##   using the package animation
library(animation)
library(ggplot2)
x <- seq(from = 0, to = 4 * pi, by = 0.01)
y <- sin(x)
m <- data.frame(x, y)
p <- ggplot(data = m, aes(x = m$x, y = m$y))
ani.options(interval = 0.02)
for (i in 1:100) {
    m$y <- sin(m$x + i)
    k <- p + geom_point(color = "pink")
    print(k)
    ani.pause()
}
